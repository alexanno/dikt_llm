import matplotlib.pyplot as plt
import numpy as np

# Generer data for figuren
x = np.linspace(0, 10, 100)
y = np.sin(x) + np.cos(x)

# Lag figuren
plt.figure(figsize=(8, 6))
plt.plot(x, y, label='Sinus + Cosinus')
plt.title('En Grå Sko - En Matplotlib Figur')
plt.xlabel('X-akse')
plt.ylabel('Y-akse')
plt.grid(True)
